# Input

Displays a form input.

<Code code={`import { Input } from '@nerdfish/ui'`} />

## Properties

<Properties>
  <Property name="type" type="string">
    The type of input to display. Defaults to `text`.
  </Property>
  <Property name="placeholder" type="string">
    The placeholder text to display.
  </Property>
  <Property name="defaultValue" type="string">
    The default value of the input.
  </Property>
  <Property name="description" type="string">
    The description to display.
  </Property>
  <Property name="error" type="string">
    The error message to display.
  </Property>
  <Property name="label" type="boolean">
    The label to display.
  </Property>
  <Property name="inputSize" type="['sm','md','lg']">
    The size of the input. Defaults to `md`.
  </Property>
</Properties>
