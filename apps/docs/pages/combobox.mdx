## Properties

<Properties>
  <Property name="items" type="Array" required>
    An array of objects with a value and label property.
  </Property>
  <Property name="label" type="String" required>
    The label of the combobox.
  </Property>
  <Property name="placeholder" type="String">
    The placeholder of the combobox.
  </Property>
  <Property name="emptyString" type="String">
    The string that is displayed when no items are found.
  </Property>
</Properties>
