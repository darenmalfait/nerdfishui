## Properties

<Properties>
  <Property name="variant" type={['danger', 'warning', 'success', 'info']}>
    Preset color scheme for the alert. Defaults to `info`.
  </Property>
  <Property name="title" type="string">
    Title of the alert.
  </Property>
  <Property name="description" type="React.ReactNode">
    Description of the alert.
  </Property>
  <Property name="hideIcon" type="boolean">
    Hide the icon of the alert.
  </Property>
</Properties>
