## Usage

```tsx
import { NavigationMenu } from '@nerdfish/ui'
```

```tsx
<NavigationMenu>
	<NavigationMenu.List>
		<NavigationMenu.Item>
			<NavigationMenu.Trigger>Item One</NavigationMenu.Trigger>
			<NavigationMenu.Content>
				<NavigationMenu.Link>Link</NavigationMenu.Link>
			</NavigationMenu.Content>
		</NavigationMenu.Item>
	</NavigationMenu.List>
</NavigationMenu>
```
