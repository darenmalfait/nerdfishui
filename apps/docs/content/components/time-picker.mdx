## Usage

```tsx
import {
	TimePicker,
	TimePickerSegment,
	TimePickerSeparator,
} from '@nerdfish/ui'
```

```tsx
<TimePicker defaultValue={new Date()}>
	<TimePickerSegment segment="hours" />
	<TimePickerSeparator>:</TimePickerSeparator>
	<TimePickerSegment segment="minutes" />
</TimePicker>
```
