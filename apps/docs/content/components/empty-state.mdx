## Usage

```tsx
import {
	EmptyState,
	Button,
	EmptyStateIcon,
	EmptyStateTitle,
	EmptyStateDescription,
	EmptyStateActions,
} from '@nerdfish/ui'
import { TriangleAlertIcon } from 'lucide-react'
```

```tsx
<EmptyState>
	<EmptyStateIcon>
		<TriangleAlertIcon />
	</EmptyStateIcon>
	<EmptyStateTitle>No data</EmptyStateTitle>
	<EmptyStateDescription>No data to display</EmptyStateDescription>
	<EmptyStateActions>
		<Button>Add data</Button>
	</EmptyStateActions>
</EmptyState>
```
