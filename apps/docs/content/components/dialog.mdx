## Usage

```tsx
import {Dialog} from '@nerdfish/ui'
```

```tsx
<Dialog>
  <Dialog.Trigger>Remove order</Dialog.Trigger>
  <Dialog.Content>
    <Dialog.Header>
      <Dialog.Title>Are you sure absolutely sure?</Dialog.Title>
      <Dialog.Description>
        This action cannot be undone. This will permanently delete your cake
        order and remove your dessert from our kitchen.
      </Dialog.Description>
    </Dialog.Header>
  </Dialog.Content>
</Dialog>
```
