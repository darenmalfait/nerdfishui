## Usage

```tsx
import {ProgressiveImage} from '@nerdfish/ui'
```

```tsx
<ProgressiveImage
  className="w-full"
  img={<img src="https://picsum.photos/3000/3000" alt="picsum" />}
  placeholder="https://picsum.photos/25/25"
/>
```
