## Usage

```tsx
import { Avatar } from '@nerdfish/ui'
import Image from 'next/image'
```

```tsx
<Avatar.Root>
	<Avatar.Image
		src="https://avatars.githubusercontent.com/u/56068461?s=40&v=4"
		alt="@darenmalfait"
	/>
	<Avatar.Fallback>DM</Avatar.Fallback>
</Avatar.Root>
```
