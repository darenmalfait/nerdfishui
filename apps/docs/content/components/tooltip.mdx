## Usage

```tsx
import {
	Tooltip,
	TooltipProvider,
	TooltipTrigger,
	TooltipContent,
} from '@nerdfish/ui'
```

```tsx
<TooltipProvider>
	<Tooltip>
		<TooltipTrigger>Hover</TooltipTrigger>
		<TooltipContent>
			<p>Add to library</p>
		</TooltipContent>
	</Tooltip>
</TooltipProvider>
```
