## Usage

```tsx
import { Mockup } from '@nerdfish/ui'
```

```tsx
<Mockup.Root>
	<Mockup.Window>
		<pre data-prefix="1">
			<code>npm i @nerdfish/ui</code>
		</pre>
		<pre data-prefix="2">
			<code>installing...</code>
		</pre>
		<pre data-prefix="3" className="bg-warning text-warning">
			<code>Error!</code>
		</pre>
	</Mockup.Window>
</Mockup.Root>
```
