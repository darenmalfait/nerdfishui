## Usage

```tsx
import {
	NavigationMenu,
	NavigationMenuList,
	NavigationMenuItem,
	NavigationMenuTrigger,
	NavigationMenuContent,
	NavigationMenuLink,
} from '@nerdfish/ui'
```

```tsx
<NavigationMenu>
	<NavigationMenuList>
		<NavigationMenuItem>
			<NavigationMenuTrigger>Item One</NavigationMenuTrigger>
			<NavigationMenuContent>
				<NavigationMenuLink>Link</NavigationMenuLink>
			</NavigationMenuContent>
		</NavigationMenuItem>
	</NavigationMenuList>
</NavigationMenu>
```
