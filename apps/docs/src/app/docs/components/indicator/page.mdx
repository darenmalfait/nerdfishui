## Usage

```tsx
import { Indicator, IndicatorItem, Badge } from '@nerdfish/ui'
```

```tsx
<Indicator>
	<IndicatorItem>
		<Badge variant="secondary" />
	</IndicatorItem>
	<div className="bg-inverted text-inverted grid h-32 w-32 place-items-center">
		content
	</div>
</Indicator>
```
