## Usage

```tsx
import { Alert, AlertTitle, AlertDescription } from '@nerdfish/ui'
```

```tsx
<Alert variant="danger">
	<AlertTitle>Example alert</AlertTitle>
	<AlertDescription>Example description</AlertDescription>
</Alert>
```
