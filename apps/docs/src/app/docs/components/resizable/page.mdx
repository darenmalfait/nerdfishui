## Usage

```tsx
import { Resizable, ResizablePanel, ResizableHandle } from '@nerdfish/ui'
```

```tsx
<Resizable
	direction="horizontal"
	className="rounded-container max-w-md border md:min-w-[450px]"
>
	<ResizablePanel defaultSize={50}>
		<div className="p-md flex h-[200px] items-center justify-center">
			<span className="font-semibold">One</span>
		</div>
	</ResizablePanel>
	<ResizableHandle />
	<ResizablePanel defaultSize={50}>
		<Resizable direction="vertical">
			<ResizablePanel defaultSize={25}>
				<div className="p-md flex h-full items-center justify-center">
					<span className="font-semibold">Two</span>
				</div>
			</ResizablePanel>
			<ResizableHandle />
			<ResizablePanel defaultSize={75}>
				<div className="p-md flex h-full items-center justify-center">
					<span className="font-semibold">Three</span>
				</div>
			</ResizablePanel>
		</Resizable>
	</ResizablePanel>
</Resizable>
```
