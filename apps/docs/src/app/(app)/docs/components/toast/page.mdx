## Toaster

This needs to be placed in the root component of your application.

```tsx
import { Toaster } from '@nerdfish/react/toast'

function RootLayout({ children }: RootLayoutProps) {
	return (
		<html lang="en">
			<body>
				<Toaster />
				{children}
			</body>
		</html>
	)
}
```
